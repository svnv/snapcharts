Snapcharts

Snapcharts is a small open source charting library built on top of snap.svg. It is currently a work in progress and should not be used by anyone for anything yet.

You can find the github repository for snapchat here.

Snapcharts is available under an Apache 2 license.

Bar chart

A simple bar chart.

        
  var options = {
    type:'bar',
    labels:['jan','feb','mar','apr','may','jun','jul','aug'],
    data:[{
      values: [.90,.20,.40,.35,.37,.50,.50,.50],
      color: 'rgba(0,175,255,.5)'
    }]
  }
  var chart = SnapChart('#bar-chart',options);

Pie chart

A simple pie chart.

        
  var options = {
    height:250,
    width:500,
    type:'pie',
    data:[{
      values: [.65],
      color: 'rgba(0,175,255,.5)',
      name: 'Segment A'
    },
    {
      values: [.35],
      color: 'rgba(175,255,0,.5)',
      name: 'Segment B'
    }]
  }
  var chart = SnapChart('#pie-chart',options);
